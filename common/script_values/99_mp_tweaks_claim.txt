claim_piety_value = {
	value = {
		add = minor_piety_value
		multiply = 2
		if = {
			limit = {
				scope:target = {
					tier = tier_duchy
				}
			}
			multiply = 2
		}
		if = {
			limit = {
				scope:target = {
					tier = tier_kingdom
				}
			}
			multiply = 4
		}
		if = {
			limit = {
				scope:target = {
					tier = tier_empire
				}
			}
			multiply = 8
		}

		# added by viegnar
		if = {
			limit = {
				scope:actor = {
					is_ai = no
				}
			}
			multiply = 1000
		}

		add = {
			value = scope:actor.sub_realm_size
			multiply = 10
		}
	}
}
